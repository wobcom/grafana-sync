1use aes_gcm::aead::consts::U12;
2use aes_gcm::aead::{Aead, Nonce, OsRng};
3use aes_gcm::aes::Aes256;
4use aes_gcm::{AeadCore, Aes256Gcm, AesGcm, KeyInit};
5use std::fmt::{Debug, Formatter};
6use tracing::instrument;
7
8#[derive(Clone)]
9pub struct EncryptedCredential {
10    cred: Vec<u8>,
11    cipher: Aes256Gcm,
12    nonce: Nonce<AesGcm<Aes256, U12>>,
13}
14
15// Because of the guarantee that a String is valid UTF-8, except is used since other cases would be UB
16impl EncryptedCredential {
17    #[instrument]
18    pub fn new(credential: String) -> EncryptedCredential {
19        let key = Aes256Gcm::generate_key(OsRng);
20        let cipher = Aes256Gcm::new(&key);
21        let nonce = Aes256Gcm::generate_nonce(OsRng);
22        let ciphertext = cipher
23            .encrypt(&nonce, credential.as_bytes())
24            .expect("encryption failure!");
25
26        EncryptedCredential {
27            cred: ciphertext,
28            cipher,
29            nonce,
30        }
31    }
32
33    #[instrument]
34    pub fn value(&self) -> String {
35        let plaintext = self
36            .cipher
37            .decrypt(&self.nonce, self.cred.as_slice())
38            .expect("decryption failure!");
39        String::from_utf8(plaintext).expect("decryption failure!")
40    }
41
42    #[instrument]
43    pub fn checkable_obfuscated(&self) -> String {
44        self.value()
45            .chars()
46            .enumerate()
47            .map(|(i, c)| {
48                if i < 4 || !c.is_alphanumeric() {
49                    c
50                } else {
51                    'x'
52                }
53            })
54            .collect()
55    }
56}
57
58impl Debug for EncryptedCredential {
59    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
60        write!(f, "{:?}", self.cred)
61    }
62}
63
64impl From<String> for EncryptedCredential {
65    #[instrument]
66    fn from(value: String) -> Self {
67        Self::new(value)
68    }
